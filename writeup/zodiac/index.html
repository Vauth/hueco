import hashlib

def Wordlist():
    with open("words.txt", "r") as f: return f.read().splitlines()

def Crack(hash_password, salt):
    for word in Wordlist():
        if hashlib.md5((salt + word.strip()).encode()).hexdigest() == hash_password: return word
    return None

# Example usage

function generateUUID($dateString) {

    // Format: Mon, 17 Jul 1954 12:34:21 GMT
    $dateTime = new DateTime($dateString, new DateTimeZone('GMT'));
    $time = $dateTime->format('U'); // Unix timestamp

    // Calculate the UUID timestamp
    $timestamp = ($time * 10000000) + 0x01B21DD213814000;

    // Get the time-related fields
    $time_low = $timestamp & 0xFFFFFFFF;
    $time_mid = ($timestamp >> 32) & 0xFFFF;
    $time_hi_and_version = ($timestamp >> 48) & 0x0FFF;
    $time_hi_and_version |= (1 << 12); // Set the version to 1
